Xét các khẳng định sau:
\begin{SubQuestions}
  \TFItem{true}{Đồ thị cắt trục hoành tại hai điểm có hoành độ là $-1$ và $3$.}
  \TFItem{true}{Trục đối xứng của parabol là đường thẳng $x=1$.}
  \TFItem{false}{Giá trị lớn nhất của hàm số bằng $4$.}
  \TFItem{true}{Bất phương trình $f(x)\le 0$ có tập nghiệm là $[-1;3]$.}
\end{SubQuestions}
\SolutionBlock{Từ đồ thị, parabol mở lên, có hai nghiệm $x=-1$ và $x=3$, đỉnh là $(1;-4)$. Vậy a), b), d) đúng; c) sai vì giá trị lớn nhất không tồn tại, còn giá trị nhỏ nhất là $-4$.}

\NQuestion Trong mặt phẳng tọa độ $Oxy$, cho đường thẳng $d:x-2y+3=0$ và điểm $A(1;2)$. Xét các khẳng định sau:
\begin{SubQuestions}
  \TFItem{true}{Một vectơ pháp tuyến của $d$ là $\vec n=(1;-2)$.}
  \TFItem{false}{Điểm $A$ thuộc đường thẳng $d$.}
  \TFItem{true}{Đường thẳng đi qua $A$ và song song với $d$ có phương trình $x-2y+3=0$.}
  \TFItem{false}{Khoảng cách từ $A$ đến $d$ bằng $0$.}
\end{SubQuestions}
\SolutionBlock{Đường thẳng $d$ có một vectơ pháp tuyến là $(1;-2)$ nên a) đúng. Thay $A(1;2)$ vào $d$ được $1-4+3=0$, do đó $A$ thuộc $d$, vì thế b) sai. Đường thẳng đi qua $A$ và song song với $d$ chính là $d$, nên c) đúng. Khoảng cách từ một điểm nằm trên đường thẳng đến đường thẳng bằng $0$, vì vậy mệnh đề d) sai.}
